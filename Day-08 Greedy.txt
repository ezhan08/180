1. N meetings in one room

       int maxMeetings(int start[], int end[], int n)
    {
        // Your code here
        vector<vector<int>> meet;
        for(int i=0; i<n;i++){
            meet.push_back({end[i], start[i]});
        }
        sort(meet.begin(), meet.end());
        
        int cnt=0, free=0;
        for(int i=0;i<n;i++){
            if(meet[i][1] > free){
                cnt++;
                free = meet[i][0];
            }
        }
        return cnt;
    }
