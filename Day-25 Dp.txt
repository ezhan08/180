class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        res = max(nums)
        maxp, minp = 1,1
        
        for n in nums:
            if n==0:
                maxp, minp = 1, 1
                continue
            tmp = n * maxp    
            maxp = max(n , n*maxp, n*minp)
            minp = min(n , tmp, n*minp)
            res = max(res, maxp)
        return res
